# create_project.py
# Create a new isolated project in Amazon Bedrock

import boto3
import json

bedrock = boto3.client(
    "bedrock",
    region_name="us-east-1"
)

# Create a project for your production ML team
response = bedrock.create_project(
    name="prod-ml-team",
    description="Production inference for ML Platform team",
    tags={
        "team":        "ml-platform",
        "env":         "production",
        "cost-center": "CC-1042",
        "owner":       "platform-eng@company.com"
    }
)

project_id = response["project"]["projectId"]
project_arn = response["project"]["projectArn"]

print(f"Created project: {project_id}")
print(f"ARN: {project_arn}")

# Drop-in: use OpenAI SDK pointing at Bedrock
# Just swap the base_url + add project header

from openai import OpenAI
import boto3, os

session = boto3.Session()
creds   = session.get_credentials().get_frozen_credentials()

client = OpenAI(
    base_url="https://bedrock-mantle.us-east-1.amazonaws.com/v1",
    api_key=creds.token,
    default_headers={
        # ← Scope every request to this project
        "X-Bedrock-Project-Id": "prod-ml-team"
    }
)

resp = client.chat.completions.create(
    model="anthropic.claude-3-5-sonnet-20241022-v2:0",
    messages=[
        {"role": "system",  "content": "You are a helpful assistant."},
        {"role": "user",    "content": "Summarize Q4 earnings."}
    ],
    max_tokens=512
)

print(resp.choices[0].message.content)

3 — Multi-Project Management
multi_project_manager.py
Python · boto3
# Provision and manage projects across teams/environments
import boto3
from dataclasses import dataclass
from typing import List

@dataclass
class ProjectSpec:
    name:        str
    env:         str          # production | staging | dev
    team:        str
    cost_center: str
    iam_role:    str

class BedrockProjectManager:
    def __init__(self, region: str = "us-east-1"):
        self.client = boto3.client("bedrock", region_name=region)

    def provision_projects(self, specs: List[ProjectSpec]) -> dict:
        created = {}
        for spec in specs:
            resp = self.client.create_project(
                name=spec.name,
                tags={
                    "env":         spec.env,
                    "team":        spec.team,
                    "cost-center": spec.cost_center,
                    "iam-role":    spec.iam_role
                }
            )
            created[spec.name] = resp["project"]["projectId"]
            print(f"  ✓ {spec.name} [{spec.env}] → {created[spec.name]}")
        return created

    def list_projects(self, env_filter: str = None):
        projects = self.client.list_projects()["projects"]
        if env_filter:
            projects = [p for p in projects
                        if p.get("tags", {}).get("env") == env_filter]
        return projects

# ── Bootstrap all environments ──────────────────────────────────────
mgr = BedrockProjectManager()
mgr.provision_projects([
    ProjectSpec("prod-ml-team",    "production", "ml-platform", "CC-1042", "arn:aws:iam::123:role/ProdMLRole"),
    ProjectSpec("staging-nlp",     "staging",    "nlp-research", "CC-2011", "arn:aws:iam::123:role/StagingNLPRole"),
    ProjectSpec("dev-experiments", "dev",        "all-engineers","CC-9001", "arn:aws:iam::123:role/DevRole"),
])
4 — TypeScript · Node.js Client
bedrockClient.ts
TypeScript
// bedrockClient.ts — project-scoped OpenAI client
import OpenAI from "openai"
import { fromNodeProviderChain } from "@aws-sdk/credential-providers"

const createProjectClient = async (
  projectId: string
): Promise<OpenAI> => {
  const creds = await fromNodeProviderChain()()

  return new OpenAI({
    baseURL: "https://bedrock-mantle.us-east-1.amazonaws.com/v1",
    apiKey:  creds.sessionToken ?? "",
    defaultHeaders: {
      "X-Bedrock-Project-Id": projectId,
      "X-Amz-Security-Token": creds.sessionToken ?? "",
    }
  })
}

export { createProjectClient }

// inference.ts — use the scoped client
import { createProjectClient } from "./bedrockClient"

async function run() {
  const client = await createProjectClient("prod-ml-team")

  const stream = await client.chat.completions.create({
    model:    "anthropic.claude-3-5-sonnet-20241022-v2:0",
    stream:   true,
    messages: [
      { role: "system", content: "You are a code reviewer." },
      { role: "user",   content: "Review this PR diff: ..."  }
    ]
  })

  for await (const chunk of stream) {
    process.stdout.write(
      chunk.choices[0]?.delta?.content ?? ""
    )
  }
}

Key Points
No Extra Charge

Projects API is available at no additional cost. You pay only for the underlying model inference tokens consumed — identical to non-project usage.

OpenAI Migration

The Projects API is fully OpenAI-compatible. Existing apps using the OpenAI SDK require only a base_url swap and a project header to migrate to Bedrock.

Available APIs

Projects scope is supported across both the Responses API and the Chat Completions API, both served through the Mantle distributed inference engine.
